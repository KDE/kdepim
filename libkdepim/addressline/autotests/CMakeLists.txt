macro(add_libkdepim_addressline_unittest _name)
    ecm_add_test(${ARGN}
        TEST_NAME ${_name}
        NAME_PREFIX "libkdepim-addressline-"
        LINK_LIBRARIES Qt5::Test KF5::Contacts Qt5::Widgets kdepim Qt5::DBus
    )
endmacro()
