kde4_header()

add_subdirectory( pics ) 

include_directories( ${CMAKE_SOURCE_DIR}/ktnef ${KDE4_INCLUDE_DIR} ${QT_INCLUDES} ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR}  )


########### next target ###############

set(ktnef_bin_SRCS 
   ktnefmain.cpp 
   ktnefview.cpp 
   attachpropertydialog.cpp 
   messagepropertydialog.cpp 
   main.cpp )

kde4_automoc(${ktnef_bin_SRCS})

kde4_add_ui3_files(ktnef_bin_SRCS attachpropertydialogbase.ui )

kde4_add_executable(ktnef_bin ${ktnef_bin_SRCS})
set_target_properties(ktnef_bin PROPERTIES OUTPUT_NAME ktnef)

target_link_libraries(ktnef_bin  ${KDE4_KDEUI_LIBRARIES} ktnef )

install_targets(/bin ktnef_bin )


########### install files ###############

install_files( ${XDG_APPS_DIR} FILES ktnef.desktop )
install_files( ${DATA_INSTALL_DIR}/ktnef FILES ktnefui.rc )
install_files( ${MIME_INSTALL_DIR}/application FILES ms-tnef.desktop )

kde4_footer()
